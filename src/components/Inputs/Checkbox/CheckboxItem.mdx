# Checkbox.Item Themed

> A clickable UI element with active and hover states. Indicates to the user that it's clickable.

<CodeBlock defaultValue={true}>
  {(value, onChange) => (
    <Checkbox.Item themed label="Node Type" value={value} onChange={onChange} />
  )}
</CodeBlock>

## Examples

<CodeBlock title="Disabled" defaultValue={false}>
  {(value, onChange) => (
    <Checkbox.Item
      label="Node Type"
      value={value}
      onChange={onChange}
      themed
      disabled
    />
  )}
</CodeBlock>

<CodeBlock title="Disabled checked" defaultValue={true}>
  {(value, onChange) => (
    <Checkbox.Item
      label="Node Type"
      value={value}
      onChange={onChange}
      themed
      disabled
    />
  )}
</CodeBlock>

<CodeBlock title="OnChange Callback" defaultValue={false}>
  {(value, onChange) => (
    <Checkbox.Item
      label="Node Type"
      value={value}
      onChange={(result) => {
        alert(JSON.stringify(result));
        onChange(result)}
      }
      themed
    />
  )}
</CodeBlock>
