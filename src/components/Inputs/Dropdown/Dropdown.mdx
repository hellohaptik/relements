# Dropdown

> A regular input element where user inputs a Number

<CodeBlock>
{
    (value, setValue) => (
        <Dropdown
            placeholder="Test placeholder"
            className="my-dropdown"
            prefixClassName="my-yoooo"
            options={[
            { text: "Option text 1" },
            { text: "Option text 2" },
            { text: "Option text 3" },
            { text: "Option text 4" },
                ]}
            label="This is the label"
            value={value}
            onChange={setValue}
        />
    )
}

</CodeBlock>
