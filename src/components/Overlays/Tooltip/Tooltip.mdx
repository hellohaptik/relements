# Tooltip

> A clickable UI element with active and hover states. Indicates to the user that it's clickable.
> Adheres to Haptik Design System.

<div
  style={{
    display: "grid",
    gridTemplateColumns: "50% 48%",
    gridColumnGap: "15px",
    marginTop: "3rem"
  }}
>

<CodeBlock
  codeString={`<div>
      <Button
        themed
        variant="secondary"
        innerRef={ref}
        onClick={...}
      >
        Open Themed Tooltip
      </Button>
      <Tooltip
        attachTo={ref}
        themed
        onClose={...}
        active={isActive}
      >
        <div>This is how I will look.</div>
      </Tooltip>
    </div>`}
>
  {(value, setValue, ref) => (
    <div>
      <Button
        themed
        variant="secondary"
        innerRef={ref}
        onClick={() => setValue(true)}
      >
        Open Themed Tooltip
      </Button>
      <Tooltip
        attachTo={ref}
        themed
        onClose={() => setValue(false)}
        active={value}
      >
        <div>This is how I will look.</div>
      </Tooltip>
    </div>
  )}
</CodeBlock>

<CodeBlock
  codeString={`<div>

        onClick={...}
      >

### On hover

> <strong>Note</strong>: This will work only with <strong>themed</strong> and <strong>toggleWithHover</strong> props.

    themed
    toggleWithHover
    tooltip="Nice!"
    variant="danger"
    position="RIGHT"
  >
    <span>Hover on me Bro!</span>
  </Tooltip>
</CodeBlock>
